452   private def awaitEndpointCompletion(endpointName : String) : Unit = {
453     val startTime = this.timeProvider.currentTimeMillis
454     val describeEndpointRequest = new DescribeEndpointRequest().withEndpointName(endpointName)
455     InternalUtils.applyUserAgent(describeEndpointRequest)
456     log.info(s"Begin waiting for endpoint completion for endpoint $endpointName")
457     while (this.timeProvider.currentTimeMillis - startTime <
458       SageMakerModel.EndpointCreationTimeout.toMillis) {
459       try {
460         val response = sagemakerClient.describeEndpoint(describeEndpointRequest)
461         val currentStatus = EndpointStatus.fromValue(response.getEndpointStatus)
462         log.info(s"Endpoint creation status: $currentStatus")
463         currentStatus match {
464           case EndpointStatus.InService => return
465           case EndpointStatus.Failed =>
466             val message = s"Endpoint '$endpointName' failed for reason:" +
467               s" '${response.getFailureReason}'"
468             throw new RuntimeException(message)
469           case _ => // for any other statuses, continue polling
470         }
471       } catch {
472         case e : SdkBaseException =>
473           if (!RetryUtils.isRetryableServiceException(e)) {
474             throw e
475           }
476           log.warn(s"Retryable exception: ${e.getMessage}")
477         case t : Throwable => throw t
478       }
479 
480       timeProvider.sleep(SageMakerModel.EndpointPollInterval.toMillis)
481     }
482 
483     throw new RuntimeException(
484       s"Timed out after ${SageMakerModel.EndpointCreationTimeout.toString}" +
485       s" while waiting for Endpoint '$endpointName' to finish creating.")
486   }

499     if (endpointCreationPolicy == EndpointCreationPolicy.CREATE_ON_TRANSFORM &&

502     }
503     val dataFrame = dataset.toDF()

509     val resultDF = dataFrame.mapPartitions(RequestBatchIteratorFactory.createRequestBatchIterator

511         responseRowDeserializer, prependResultRows))(RowEncoder(resultSchema))
512 
513     resultDF
514   }
