27 /**
28   * Uploads a DataFrame to S3 using the given data format and with the given options. Detects
29   * if the application is being run on an EMR cluster to determine the S3 FileSystem scheme,
30   * and whether to specify Training input data using a manifest file rather than an
31   * S3 Prefix.
32   *
33   * [[com.amazonaws.services.sagemaker.sparksdk.SageMakerEstimator]]
34   * depends on DataUploader to write DataFrames to S3 for training on
35   * Amazon SageMaker.
36   *
37   * @param dataFormat the Spark-supported data format to write the DataFrame as, such as
38   *                   "sagemaker" or "libsvm"
39   * @param dataFormatOptions a map of options to configure the Spark DataFrameWriter corresponding
40   *                          to the dataFormat.
41   */
42 private[sparksdk] class DataUploader(final val dataFormat: String, final val dataFormatOptions:
43 collection.immutable.Map[String, String]) {
44 
45   /* Prefix for local file system temporary files containing a manifest of SageMaker training
46   files */
47   private final val ManifestTempFilePrefix = "sagemaker-manifest"
48 
49   /* Suffix for manifest s3 objects */
50   private final val ManifestKeySuffix = ".manifest.txt"
51 
52   /* S3 metadata object key suffix, created by the hadoop filesystem */
53   private final val HDFSFolderSuffix = "$folder$"
54 
55   /* Hadoop file system entry used to indicate success of a map-reduce file output committer */
56   private final val HadoopSuccessFile = "_SUCCESS"
57 
58   /* Filesystem scheme to use when uploading to S3. */
59   private final val S3FSScheme = "s3a"
60 
61   /* Filesystem scheme to use when uploading to S3 from EMR */
62   private final val EMRFSScheme = "s3"
63 
64   /* Hadoop property that names the Filesystem implementation class. */
65   private final val EmrfsProperty = "fs.s3.impl"
66 
67   /* When running on EMR, the property "fs.s3.impl" is this class name. */
68   private final val EmrFSClassName = "EmrFileSystem"
69 
70 
71   /**
72     * Uploads a dataset to S3. If running on EMRFS, also uploads a manifest file containing a
73     * list of the uploaded files.
74     *
75     * @param inputS3DataPath S3 URI to upload dataset to
76     * @param dataset         Dataset to upload
77     * @return a DataUploadResult containing an S3Path, either a ManifestDataUploadResult with an
78     *         S3DataPath to a manifest file if a manifest was uploaded to S3 or an
79     *         ObjectPrefixUploadResult with an S3DataPath to the S3 location
80     *         where the dataset was uploaded if a manifest was not uploaded to S3.
81     */
82   def uploadData(inputS3DataPath: S3DataPath, dataset: Dataset[_]): DataUploadResult = {
83     val inputDataBucket = inputS3DataPath.bucket
84     val inputDataObjectPath = inputS3DataPath.objectPath
85 
86     val onEMR = usingEMRFS(dataset)
87     val fsScheme = getFSScheme(onEMR)
88 
89     val inputURI = s"$fsScheme://$inputDataBucket/$inputDataObjectPath"
90     writeData(dataset, inputURI)
91 
92     if (onEMR) {
93       // Use manifest
94       val inputPath = new Path(inputURI)
95       val hadoopFS = getHadoopFilesystem(inputPath, dataset)
96       val manifestKey = writeAndUploadManifest(inputPath, hadoopFS)
97       ManifestDataUploadResult(S3DataPath(s"$inputDataBucket", s"$manifestKey"))
98     } else {
99       ObjectPrefixUploadResult(S3DataPath(inputDataBucket, inputDataObjectPath))
100     }
101   }
102 
103   private[internal] def getFSScheme(onEMR: Boolean) : String = {
104     if (onEMR) EMRFSScheme else S3FSScheme
105   }
106 
107   private[sparksdk] def writeData(dataset: Dataset[_], inputURI: String): Unit = {
108     // Do not write _SUCCESS file.
109     dataset.sparkSession.sparkContext.hadoopConfiguration.set(
110       "mapreduce.fileoutputcommitter.marksuccessfuljobs", "false")
111     dataset.write.format(dataFormat).options(dataFormatOptions).save(inputURI)
112   }
113 
114   private[internal] def usingEMRFS(dataset: Dataset[_]): Boolean = {
115     val fileSystemClassName = dataset.sparkSession.sparkContext.
116       hadoopConfiguration.get(EmrfsProperty)
117     fileSystemClassName != null && fileSystemClassName.contains(EmrFSClassName)
118   }
119 
120   private[internal] def getHadoopFilesystem(inputPath: Path, dataset: Dataset[_]): FileSystem = {
121     inputPath.getFileSystem(dataset.sparkSession.sparkContext.hadoopConfiguration)
122   }
123 
124   private[internal] def writeAndUploadManifest(sagemakerInput: Path, hadoopFS: FileSystem):
125   String = {
126     val tempFile = File.createTempFile(ManifestTempFilePrefix, ManifestTempFilePrefix)
127     val manifestWriter = new FileWriter(tempFile)
128     try {
129       writeManifest(sagemakerInput, hadoopFS, manifestWriter)
130     } finally {
131       manifestWriter.close()
132     }
133     val manifestPath = new Path(s"${sagemakerInput.toUri.getPath}$ManifestKeySuffix")
134     hadoopFS.copyFromLocalFile(true, new Path(tempFile.getAbsolutePath), manifestPath)
135     manifestPath.toUri.getPath.stripPrefix("/") // Remove leading '/' to produce an s3 key
136   }
137 
138   /**
139     * Writes manifest entries for sagemaker training input files rooted in the input path.
140     *
141     * @param sagemakerInput The input path containing sagemaker data files
142     * @param hadoopFS       The hadoop file system
143     * @param manifestWriter The writer to write manifest entries to
144     */
145   private[internal] def writeManifest(sagemakerInput: Path,
146                                       hadoopFS: FileSystem,
147                                       manifestWriter: Writer): Unit = {
148     val hadoopFSIterator = hadoopFS.listFiles(sagemakerInput, true)
149     val manifestStringBuilder = new StringBuilder("[{\"prefix\": \"" + sagemakerInput + "/\"}, ")
150     require(hadoopFSIterator.hasNext, s"No files found at $sagemakerInput.")
151     val fileNames = ListBuffer.empty[String]
152     while (hadoopFSIterator.hasNext) {
153       val nextFile = hadoopFSIterator.next.getPath
154       val fileName = nextFile.getName
155       if (!isHadoopSpecialFile(fileName)) {
156         fileNames.append("\"" + fileName + "\"")
157       }
158     }
159     require(fileNames.nonEmpty, s"No non-hadoop files found at $sagemakerInput.")
160     manifestStringBuilder.append(fileNames.mkString(", "))
161     manifestWriter.write(manifestStringBuilder.append("]").toString)
162   }
163 
164   private[internal] def isHadoopSpecialFile(fileName: String): Boolean = {
165     fileName.endsWith(HDFSFolderSuffix) || fileName.equals(HadoopSuccessFile)
166   }
167 }
