33 /**
34   * A [[ResponseRowDeserializer]] for converting SageMaker Protobuf-in-recordio response data
35   * to Spark rows.
36   * @param schema The schema of rows in the response.
37   * @param protobufKeys The keys for each output in the Protobuf response.
38   */
39 class ProtobufResponseRowDeserializer(override val schema: StructType,
40                                       val protobufKeys : Option[Seq[String]] = None)
41   extends ResponseRowDeserializer {
42 
43   /**
44     * Deserialize a SageMaker response containing a byte array of record-IO encoded SageMaker
45     * Protobuf records.
46     *
47     * @param responseData The Array[Byte] containing the Protobuf-in-recordIO response
48     * @return An Iterator over deserialized Rows
49     */
50   override def deserializeResponse(responseData: Array[Byte]): Iterator[Row] = {
51     val recordIterator = ProtobufConverter.recordIOByteArrayToProtobufs(responseData)
52     val rowList = ListBuffer[Row]()
53     while (recordIterator.hasNext) {
54       rowList += protobufToRow(recordIterator.next)
55     }
56     rowList.toIterator
57   }
58 
59   private def protobufToRow(record: Record) : Row = {
60     val rowValues = ListBuffer[Any]()
61     val labelList = record.getLabelList
62     for (fieldName <- protobufKeys.getOrElse[Seq[String]](schema.fieldNames)) {
63       val value = getValueFromLabelList(fieldName, labelList.asScala)
64       val dataValue = getValueFromTensor(value.getFloat32Tensor)
65       rowValues += dataValue
66     }
67     if (schema.fields.length > rowValues.toArray.length) {
68       throw new IllegalArgumentException(s"Expected ${schema.fields.length} entries to fit" +
69         s" schema ${schema}, received ${rowValues.toArray.length} entries: ${rowValues.toList}")
70     }
71     new GenericRowWithSchema(rowValues.toArray, schema)
72   }
73 
74   private def getValueFromLabelList(fieldName: String,
75                                     labelList: mutable.Buffer[MapEntry]): Value = {
76     val mapEntryList = labelList.filter(_.getKey == fieldName)
77     require(mapEntryList.nonEmpty, s"Couldn't find field $fieldName in $labelList")
78     mapEntryList.head.getValue
79   }
80 
81   private def getValueFromTensor(valuesTensor: Float32Tensor) : Any = {
82     val valuesCount = valuesTensor.getValuesCount
83     require(valuesCount > 0, "Can't get value from deserialized tensor: values list is empty.")
84 
85     if (valuesCount == 1) {
86       // Get value as a scalar
87       valuesTensor.getValues(0).toDouble
88     } else {
89       // Get value as a Vector
90       val keyCount = valuesTensor.getKeysCount
91       val values = asScalaBufferConverter(valuesTensor.getValuesList)
92         .asScala.toArray.map(_.toDouble)
93         if (keyCount > 0) {
94         // Tensor is sparsely encoded. We can only represent sparsely-encoded vectors
95         // (not higher-order tensors).
96         require(valuesTensor.getShapeCount == 1,
97           "Cannot deserialize tensor to vector. Shape list has more than one value.")
98         val indices = asScalaBufferConverter(valuesTensor.getKeysList).asScala.toArray.map(_.toInt)
99         return new SparseVector(valuesTensor.getShape(0).toInt, indices, values)
100       } else {
101         // Vector is densely encoded.
102         return new DenseVector(values)
103       }
104     }
105   }

109     */

111 }
