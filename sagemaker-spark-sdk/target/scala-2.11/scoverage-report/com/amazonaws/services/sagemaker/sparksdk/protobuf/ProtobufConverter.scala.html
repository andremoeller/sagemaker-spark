29 /**
30   * Utility functions that convert to and from the Amazon Record protobuf data format and encode
31   * Records in recordIO.
32   *
33   * @see [[https://aws.amazon.com/sagemaker/latest/dg/cdf-training.html/]] for more information on
34   *     the Amazon Record data format
35   * @see [[https://mxnet.incubator.apache.org/architecture/note_data_loading.html]] for more
36   *      information on recordIO
37   */
38 object ProtobufConverter {
39 
40   /** Certain first-party algorithms (like K-Means, PCA, and LinearLearner) on Amazon SageMaker
41     * expect each Amazon Record protobuf message to contain a "features" map with key "values"
42     * that encodes the example's features vector.
43     */
44   val ValuesIdentifierString = "values"
45 
46   /**
47     * Converts a Row to a SageMaker Protobuf record for training or inference.
48     *
49     * @param row a Row with a column of Doubles for labels and a Vector for features
50     * @param featuresFieldName index in the row corresponding to the features column
51     * @param labelFieldName index in the row corresponding to the label column
52     * @return a SageMaker Protobuf record representing the Row
53     */
54   def rowToProtobuf(row: Row, featuresFieldName: String,
55                     labelFieldName: Option[String] = Option.empty): Record = {
56     require(row.schema != null, "Row schema is null for row " + row)
57     val protobufBuilder : Builder = Record.newBuilder()
58 
59     if (labelFieldName.nonEmpty) {
60       val hasLabelColumn = row.schema.fieldNames.contains(labelFieldName.get)
61       if (hasLabelColumn) {
62         setLabel(protobufBuilder, row.getAs[Double](labelFieldName.get))
63       }
64     }
65     val hasFeaturesColumn = row.schema.fieldNames.contains(featuresFieldName)
66 
67     if (hasFeaturesColumn) {
68       val vector = row.getAs[Vector](featuresFieldName)
69       setFeatures(protobufBuilder, vector)
70     } else if (!hasFeaturesColumn) {
71       throw new IllegalArgumentException(s"Need a features column with a " +
72         s"Vector of doubles named $featuresFieldName to convert row to protobuf")
73     }
74     protobufBuilder.build
75   }

77   /**
78     * Converts a recordIO encoded byte array of Protobuf records to an iterator of Protobuf Records
79     *
80     * @see [[https://mxnet.incubator.apache.org/architecture/note_data_loading.html]] for more
81     *     information on recordIO
82     *
83     * @param byteArray recordIO encoded byte array to convert to Protobuf Record
84     * @return An Iterator over Protobuf Records
85     */
86   def recordIOByteArrayToProtobufs(byteArray: Array[Byte]) : Iterator[Record] = {
87     val recordList = new mutable.MutableList[Record]()
88     val buffer = ByteBuffer.wrap(byteArray)
89     buffer.order(ByteOrder.LITTLE_ENDIAN)
90 
91     var index = 0
92     while (buffer.hasRemaining) {
93       validateMagicNumber(buffer.getInt)
94 
95       val recordBytes = new Array[Byte](buffer.getInt)
96       buffer.get(recordBytes, 0, recordBytes.length)
97 
98       val protobufRecord = ProtobufConverter.byteArrayToProtobuf(recordBytes)
99       recordList += protobufRecord
100 
101       buffer.position(buffer.position + paddingCount(buffer.position))
102     }
103     recordList.iterator
104   }

106   private val magicNumber : Integer = 0xced7230a;
107   private val magicNumberBytes : Array[Byte] = intToLittleEndianByteArray(magicNumber)
108 
109   private def intToLittleEndianByteArray(int : Integer) : Array[Byte] = {
110     ByteBuffer.allocate(4).order(ByteOrder.LITTLE_ENDIAN).putInt(int).array
111   }

113   /**
114     * Encodes a byte array in recordIO.
115     *
116     * @see [[https://mxnet.incubator.apache.org/architecture/note_data_loading.html]] for more
117     *     information on recordIO
118     *
119     * @param byteArray Byte array to encode in recordIO
120     * @return RecordIO encoded byte array
121     */
122   def byteArrayToRecordIOEncodedByteArray(byteArray: Array[Byte]) :
123   Array[Byte] = {
124     val recordLengthBytes = intToLittleEndianByteArray(byteArray.length)
125     var recordIOBytes = magicNumberBytes ++ recordLengthBytes ++ byteArray
126     val paddingNeeded = paddingCount(recordIOBytes.length)
127     for (i <- 1 to paddingNeeded) {
128       recordIOBytes ++= Array[Byte](0)
129     }
130     recordIOBytes
131   }
132 
133   /**
134     * Returns how much padding is necessary to encode a byte array in record-IO
135     *
136     * @param byteCount length of the Record in bytes
137     * @return how many bytes to pad
138     */
139   private[protobuf] def paddingCount(byteCount: Int): Int = {
140     val mod = byteCount % 4
141     if (mod == 0) 0 else 4 - mod
142   }
143 
144   /**
145     * Validates that the given integer is equal to the RecordIO magic number that delimits records.
146     *
147     * @param recordDelimiter magic number recognizing record format
148     */
149   private[protobuf] def validateMagicNumber(recordDelimiter: Integer): Unit = {
150     if (recordDelimiter != magicNumber) {
151       throw new RuntimeException("Incorrectly encoded byte array. " +
152         "Record delimiter did not match RecordIO magic number.")
153     }
154   }

159     * @param byteArray byte array to convert to Protobuf Record

162   def byteArrayToProtobuf(byteArray: Array[Byte]): Record = {

166   private def setLabel(protobufBuilder: Record.Builder, label: Double): Record.Builder = {
167     val labelTensor = Value.newBuilder().getFloat32TensorBuilder.addValues(label.toFloat).build
168     val labelValue = Value.newBuilder().setFloat32Tensor(labelTensor).build()
169     val mapEntry = MapEntry.newBuilder().setKey(ValuesIdentifierString).setValue(labelValue).build
170     protobufBuilder.addLabel(mapEntry)
171   }
172 
173   private def setFeatures(protobufBuilder: Record.Builder,
174                                     vector: Vector): Record.Builder = {
175     val featuresTensorBuilder = Value.newBuilder().getFloat32TensorBuilder()
176 
177     val featuresTensor = vector match {
178       case v: DenseVector =>
179         for (value <- v.values) {
180           featuresTensorBuilder.addValues(value.toFloat)
181         }
182         featuresTensorBuilder.build()
183       case v: SparseVector =>
184         featuresTensorBuilder.addShape(v.size)
185         for (i <- 0 until v.indices.length) {
186           featuresTensorBuilder.addKeys(v.indices(i))
187           featuresTensorBuilder.addValues(v.values(i).toFloat)
188         }
189         featuresTensorBuilder.build()
190     }
191     val featuresValue = Value.newBuilder().setFloat32Tensor(featuresTensor).build
192     val mapEntry = MapEntry.newBuilder().setKey(ValuesIdentifierString)
193       .setValue(featuresValue).build
194     protobufBuilder.addFeatures(mapEntry)
195   }
