27 /**
28   * Deletes any SageMaker entities created during operation of the SageMaker
29   * Estimator and Transformer.
30   *
31   * @param sagemaker The AWS SDK client used to make requests to SageMaker for deleting
32   *                Training and Hosting resources.
33   */
34 class SageMakerResourceCleanup(sagemaker : AmazonSageMaker) {
35 
36   /**
37     * Deletes any SageMaker entities which have been created and still exist.
38     * Any transient exceptions occurring from service calls will be rethrown wrapped in a
39     * [[com.amazonaws.services.sagemaker.sparksdk.exceptions.SageMakerResourceCleanupException
40     * SageMakerResourceCleanupException]] and can be retried.
41     *
42     * @param resources The set of entities which have been created.
43     *                  Can be obtained from [[SageMakerModel.getCreatedResources]],
44     *                  [[SageMakerEstimator.getCreatedResources]],
45     *                  or any [[SageMakerSparkSDKException]] thrown during usage.
46     */
47   def deleteResources(resources: CreatedResources): Unit = {
48     /* Only attempt to delete resources which were created during the execution of
49      * the SageMaker Estimator/Model, to avoid accidentally deleting already existing resources
50      * with the same name.
51      *
52      * For each resource which was created, first make a Describe call.
53      * Only make the corresponding Delete call if the Describe call succeeded.
54      */
55 
56     try {
57       resources.endpointName.foreach(deleteEndpoint(_))
58       resources.endpointConfigName.foreach(deleteEndpointConfig(_))
59       resources.modelName.foreach(deleteModel(_))
60     } catch {
61       case t: Throwable => throw new SageMakerResourceCleanupException("Resource cleanup failed." +
62         " Cleanup can be retried if appropriate by calling deleteResources again.", t, resources)
63     }
64   }
65

67     val describeEndpointRequest = new DescribeEndpointRequest().withEndpointName(name)
68     InternalUtils.applyUserAgent(describeEndpointRequest)

100   /* Returns a Success if the entity exists, a Failure if it doesn't, or throws an Exception,
101    * which indicates that we aren't sure, and that this can be retried. */
102   private def tryDescribe[T](describeCall: => T): Try[T] = {
103     Try(describeCall) match {
104       case Success(v) => Success(v)
105       case Failure(e) =>
106         e match {
107           case e : AmazonSageMakerException if !RetryUtils.isRetryableServiceException(e)
108             => Failure(e)
109           case t: Throwable => throw t
110         }
111     }
112   }
