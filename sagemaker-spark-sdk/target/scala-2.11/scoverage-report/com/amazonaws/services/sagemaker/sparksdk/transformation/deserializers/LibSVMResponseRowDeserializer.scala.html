55   private def parseLibSVMRow(record: String): Row = {
56     val items = record.split(' ')
57     val label = items.head.toDouble
58     val (indices, values) = items.tail.filter(_.nonEmpty).map { item =>
59       val entry = item.split(':')

63     }.unzip
64     Row(label, new SparseVector(dim, indices.toArray, values.toArray))
65   }
